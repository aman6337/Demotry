abstract class User {
	private int userid;
	private String name;
	 // Write your code here...
     public User(int id, String n){
         this.userid=id;
         this.name=n;
     }
	 
public abstract float getDiscount(float shoppingAmt);
}
class RegisteredUser extends User{
    // Write your code here...
	 private int registrationNo;
     public RegisteredUser (int id , String n,int r){
         super (id,n);
         this.registrationNo=r;
     }

     public float getDiscount(float shoppingAmt){
         if(shoppingAmt<1000){
             return shoppingAmt;
         }else if(shoppingAmt>= 1000 && shoppingAmt<5000){
             return shoppingAmt- shoppingAmt*0.05f;
         }else if(shoppingAmt >=5000 && shoppingAmt<10000){
             return shoppingAmt - shoppingAmt*0.07f;
         }else {
             return shoppingAmt-shoppingAmt*0.10f;
         }
     }
}
class PrivilegedUser extends User {
     // Write your code here...
private String cardType;
public PrivilegedUser(int id, String n, String c){
    super(id,n);
    this.cardType=c;
}
public float getDiscount(float shoppingAmt){
    if (cardType.equals("gold")){
        return shoppingAmt-shoppingAmt*0.10f;

    }else if (cardType.equals("silver")){
        return shoppingAmt- shoppingAmt*0.03f;
    }else if(shoppingAmt>=5000){
        return shoppingAmt-shoppingAmt*0.05f;
    }
}
	 
}
